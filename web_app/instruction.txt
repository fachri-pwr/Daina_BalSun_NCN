# to dump data (geojson data) into table in database





import streamlit as st
import requests
import pandas as pd
import geopandas as gpd
from shapely.geometry import shape
from streamlit_folium import st_folium
import folium
from folium.plugins import MarkerCluster

# --- Configuration ---
st.set_page_config(layout="wide", page_title="GeoSpatial Rank Dashboard")

# CSS for the numbered markers
st.markdown("""
    <style>
    .number-icon {
        background-color: white;
        border: 2px solid #228B22;
        border-radius: 50%;
        color: black;
        font-weight: bold;
        text-align: center;
        font-size: 11px;
        line-height: 24px;
        width: 25px;
        height: 25px;
        box-shadow: 2px 2px 5px rgba(0,0,0,0.3);
    }
    </style>
""", unsafe_allow_html=True)

st.title("🌍 GeoSpatial Rank Dashboard")

API_BASE_URL = "http://127.0.0.1:8000"

# List of all score columns we want to show in the popup
SCORE_COLUMNS = ["box_id", "topsis_rank", "dni_score", "pvout_score",'dem_score','dso_score' ,"temp_score", "solar_score", "land_score",'road_score','station_score']

# --- Sidebar ---
st.sidebar.header("Data Management")
region_name = st.sidebar.text_input("Enter Region Name", value="dolnoslaskie")

if st.sidebar.button("Fetch Data from API"):
    with st.spinner(f"Downloading {region_name}..."):
        url = f"{API_BASE_URL}/geojson/all/final_score/{region_name}"
        try:
            response = requests.get(url)
            if response.status_code == 200:
                data = response.json()
                if data:
                    df = pd.DataFrame(data)
                    df = df.dropna(subset=['geometry'])
                    df['geometry'] = df['geometry'].apply(lambda x: shape(x))
                    
                    # Ensure numeric types for scores
                    for col in SCORE_COLUMNS:
                        if col in df.columns and col != "box_id":
                            df[col] = pd.to_numeric(df[col], errors='coerce')

                    gdf = gpd.GeoDataFrame(df, geometry='geometry', crs="EPSG:2180")
                    
                    # Prep coordinates for markers
                    gdf['centroid_geom'] = gdf.geometry.centroid
                    gdf = gdf.to_crs(epsg=4326)
                    gdf['lat'] = gdf['centroid_geom'].to_crs(epsg=4326).y
                    gdf['lon'] = gdf['centroid_geom'].to_crs(epsg=4326).x
                    
                    st.session_state["gdf"] = gdf
                    st.sidebar.success(f"Loaded {len(df)} rows!")
        except Exception as e:
            st.sidebar.error(f"Error: {e}")

# --- Main Logic ---
if "gdf" in st.session_state:
    gdf_web = st.session_state["gdf"]
    
    # --- Sidebar Controls ---
    st.sidebar.markdown("---")
    min_r, max_r = int(gdf_web['topsis_rank'].min()), int(gdf_web['topsis_rank'].max())
    rank_range = st.sidebar.slider("Rank Range Filter", min_r, max_r, (min_r, min(min_r+20, max_r)))
    
    # NEW CONTROLS
    show_numbers = st.sidebar.checkbox("Show All Rank Numbers", value=True)
    show_top_markers = st.sidebar.checkbox("Show Top 5 Pins (Red)", value=False)
    use_clustering = st.sidebar.checkbox("Cluster Markers", value=True)

    # Filter data based on slider
    filtered_gdf = gdf_web[
        (gdf_web['topsis_rank'] >= rank_range[0]) & 
        (gdf_web['topsis_rank'] <= rank_range[1])
    ].copy()

    # Identify Top 5 ranks WITHIN the selected slider range
    top_5_in_range = filtered_gdf.nsmallest(5, 'topsis_rank')['topsis_rank'].tolist()

    # Score Type for Map Coloring
    score_type = st.selectbox("Color Map By:", SCORE_COLUMNS[1:]) 

    # --- Map ---
    if not filtered_gdf.empty:
        # 1. Base Map with Polygons
        m = filtered_gdf.explore(
            column=score_type,
            cmap="viridis_r" if score_type == "topsis_rank" else "viridis",
            tooltip=["box_id", "topsis_rank"],
            popup=SCORE_COLUMNS,
            tiles="CartoDB positron",
            style_kwds=dict(weight=0.5, color="black")
        )

        # Container for markers
        marker_target = MarkerCluster().add_to(m) if use_clustering else m
        
        for _, row in filtered_gdf.iterrows():
            rank_val = int(row['topsis_rank'])
            
            # CASE 1: Logic for the Red Pin (Top 5 only)
            if show_top_markers and rank_val in top_5_in_range:
                marker_popup_html = f"""
                <div style="font-family: Arial; font-size: 12px; min-width: 150px;">
                    <b style="color:red;">🏆 TOP RANK #{rank_val}</b><br>
                    <hr>
                    <b>DSO Score:</b> {row['dso_score']}<br>
                    <b>Road Score:</b> {row['road_score']}<br>
                    <b>Station:</b> {row['station_score']}
                </div>
                """
                folium.Marker(
                    location=[row['lat'], row['lon']],
                    icon=folium.Icon(color='red', icon='info-sign'),
                    popup=folium.Popup(marker_popup_html, max_width=200),
                    tooltip=f"Top Rank: {rank_val}"
                ).add_to(m) # Added directly to 'm' to bypass clustering for high visibility

            # CASE 2: Logic for original Numbered Markers
            if show_numbers:
                bg_color = "#FFD700" if rank_val <= 3 else "white"
                icon_html = f'''<div class="number-icon" style="background-color: {bg_color};">{rank_val}</div>'''
                
                folium.Marker(
                    location=[row['lat'], row['lon']],
                    icon=folium.DivIcon(html=icon_html),
                    popup=f"Rank: {rank_val}"
                ).add_to(marker_target)
        
        st_folium(m, width=1200, height=600, returned_objects=[])
    else:
        st.warning("No data found for this rank range.")

    with st.expander("Raw Data Table"):
        st.dataframe(filtered_gdf[SCORE_COLUMNS].sort_values("topsis_rank"))
else:
    st.info("👈 Enter a region and fetch data to start.")
